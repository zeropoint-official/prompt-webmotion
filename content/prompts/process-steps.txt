export function ProcessSection() {
  const [activeStep, setActiveStep] = useState(0);
  const panelRefs = useRef<(HTMLElement | null)[]>([]);

  useEffect(() => {
    const observers = panelRefs.current.map((el, i) => {
      if (!el) return null;
      const obs = new IntersectionObserver(
        ([entry]) => {
          if (entry.isIntersecting) setActiveStep(i);
        },
        { threshold: 0.5 }
      );
      obs.observe(el);
      return obs;
    });
    return () => observers.forEach((o) => o?.disconnect());
  }, []);

  return (
    <section id="process" className="relative z-10 w-full bg-neutral-950 text-white">
      {/* HEADER */}
      <div className="mx-auto max-w-[1500px] px-6 pt-28 md:px-10 md:pt-40">

          <h2 className="font-sans text-[clamp(2.4rem,5.6vw,5rem)] font-medium leading-[0.98] tracking-[-0.045em] md:col-span-12">
            <span className="bg-gradient-to-r from-white to-white/60 bg-clip-text text-transparent">
              A skeleton becomes a building
            </span>
            <br />
            <span className="text-white/40">one disciplined </span>
            <span className="bg-gradient-to-r from-amber-400 to-yellow-300 bg-clip-text text-transparent">stage at a time.</span>
          </h2>
          <div className="md:col-span-6 md:col-start-7">
            <p className="mt-2 max-w-[58ch] text-[clamp(1rem,1.1vw,1.1rem)] font-light leading-[1.65] text-white/65">
              From the first survey to the final certificate, every project moves through four controlled stages. Scroll to follow one structure from raw frame to finished handover.
            </p>
          </div>
        </div>
      </div>

      {/* STICKY LAYOUT */}
      <div className="relative mx-auto mt-20 max-w-[1500px] px-6 md:mt-28 md:px-10">
        <div className="grid grid-cols-1 gap-10 md:grid-cols-12 md:gap-16">
          {/* LEFT — scrolling panels */}
          <div className="md:col-span-5">
            {steps.map((p, i) => (
              <article
                key={p.n}
                ref={(el) => { panelRefs.current[i] = el; }}
                className="flex min-h-screen flex-col justify-center py-20"
              >
                <div className="flex items-center gap-3">
                  <span className="bg-gradient-to-br from-amber-400 to-yellow-300 bg-clip-text font-sans text-[1.4rem] font-medium leading-none text-transparent">
                    {p.n}
                  </span>
                  <span className="text-[0.62rem] font-medium uppercase tracking-[0.28em] text-white/55">
                    {p.eyebrow}
                  </span>
                </div>

                <h3 className="mt-6 max-w-[16ch] font-sans text-[clamp(2rem,3.4vw,3.4rem)] font-medium leading-[0.96] tracking-[-0.035em] bg-gradient-to-r from-white to-white/70 bg-clip-text text-transparent">
                  {p.title}
                </h3>

                <p className="mt-6 max-w-[44ch] text-[1rem] font-light leading-[1.7] text-white/65">
                  {p.body}
                </p>

                <ul className="mt-8 space-y-3">
                  {p.bullets.map((b) => (
                    <li key={b} className="flex items-center gap-3 text-[0.85rem] tracking-[0.02em] text-white/70">
                      <span className="h-px w-6 bg-gradient-to-r from-amber-400 to-yellow-300" />
                      {b}
                    </li>
                  ))}
                </ul>

                {/* Mobile image */}
                <div className="relative mt-10 aspect-video w-full overflow-hidden rounded-[1.25rem] border border-white/10 md:hidden">

                </div>
              </article>
            ))}

          {/* RIGHT — sticky image */}
          <div className="hidden md:col-span-7 md:block">
            <div className="sticky top-0 flex h-screen items-center">
              <div className="relative aspect-[5/4] max-h-[90vh] w-full overflow-hidden rounded-[1.5rem] border border-white/10 bg-black/40">
                {steps.map((p, i) => (
                  <div
                    key={p.n}
                    className="absolute inset-0 transition-opacity duration-700"
                    style={{ opacity: activeStep === i ? 1 : 0, zIndex: activeStep === i ? 1 : 0 }}
                  >
                    <Image src={p.img} alt="" fill sizes="50vw" className="object-cover" priority={i === 0} />
                    <div className="absolute inset-0 bg-gradient-to-t from-black/55 via-black/10 to-transparent" />
                    <div className="absolute bottom-5 left-5 flex items-center gap-2 text-[0.6rem] uppercase tracking-[0.28em] text-white/80">

            No surprises, no shortcuts.{" "}
            <span className="text-white/40">Just a building that was always going to work.</span>
          </p>
        </div>
      </div>
    </section>
  );
}

export default ProcessSection;
